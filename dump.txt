CREATE DATABASE `test` CHARACTER SET utf8 COLLATE utf8_general_ci;
USE test;

CREATE TABLE authors (
    id serial NOT NULL PRIMARY KEY,
    name varchar(100)
);

CREATE TABLE news (
    id serial NOT NULL PRIMARY KEY,
    title varchar(100),
    content text,
    author_id BIGINT UNSIGNED,
    FOREIGN KEY (author_id) REFERENCES authors(id)
);
